    ```typescript
    // CanvasEdgeInfo -- normalized edge data
    export interface CanvasEdgeInfo {
      id: string;
      fromNode: string;  // node ID
      toNode: string;    // node ID
      label?: string;
    }

    // ProximityPair -- relationship between two nodes
    export interface ProximityPair {
      nodeA: string;  // node ID
      nodeB: string;  // node ID
      distance: number;  // Euclidean distance between centers
      relevance: number;  // 0.0-1.0, per D-02
    }

    // ProximityGraph -- all pairwise relationships
    export interface ProximityGraph {
      pairs: ProximityPair[];
      maxDistance: number;  // largest distance in the graph
      adaptiveThreshold: number;  // computed from canvas density per D-01
    }

    // Point -- simple 2D coordinate
    export interface Point {
      x: number;
      y: number;
    }

    // ViewportState -- canvas zoom and pan
    export interface ViewportState {
      x: number;      // viewport X offset in canvas coords
      y: number;      // viewport Y offset in canvas coords
      zoom: number;   // current zoom scale
    }

    // SpatialConfig -- tunable parameters (Claude's discretion values)
    export interface SpatialConfig {
      relevanceThreshold: number;      // min relevance to include in context, default 0.15
      edgeBoost: number;               // additive boost for edge-connected nodes, default 0.3
      adaptiveMultiplier: number;      // multiplier for median nearest-neighbor distance, default 2.0
      minPoints: number;               // DBSCAN minPoints parameter, default 2
      placementGap: number;            // gap between nodes in pixels, default 40
      maxSearchRadius: number;         // max outward scan multiplier, default 5
      decayFactor: number;             // exponential decay steepness, default 3.0
    }

    export const DEFAULT_SPATIAL_CONFIG: SpatialConfig = {
      relevanceThreshold: 0.15,
      edgeBoost: 0.3,
      adaptiveMultiplier: 2.0,
      minPoints: 2,
      placementGap: 40,
      maxSearchRadius: 5,
      decayFactor: 3.0,
    };
    ```

    **Step 3: Create src/spatial/proximity.ts** implementing:
    - `computeCenter(node: CanvasNodeInfo): Point` -- returns center from top-left + half dimensions. Per Pitfall 1 in RESEARCH.md, NEVER use raw x,y for distance.
    - `euclideanDistance(a: Point, b: Point): number` -- standard sqrt(dx^2 + dy^2).
    - `computeRelevance(distance: number, maxDistance: number, hasEdgeConnection: boolean, edgeBoost?: number, decayFactor?: number): number` -- exponential decay `Math.exp(-decayFactor * normalized)` where normalized = `Math.min(distance / maxDistance, 1.0)`. If hasEdgeConnection, add edgeBoost (default 0.3) and clamp to 1.0 per D-03. Round to 3 decimal places.
    - `computeAdaptiveThreshold(nodes: CanvasNodeInfo[], multiplier?: number): number` -- per D-01 and RESEARCH Pattern 2: compute nearest-neighbor distances for each node, take the median, multiply by adaptiveMultiplier (default 2.0).     - `buildProximityGraph(nodes: CanvasNodeInfo[], edges: CanvasEdgeInfo[], config?: Partial<SpatialConfig>): ProximityGraph` -- compute all pairwise distances and relevance scores. Use `computeAdaptiveThreshold` for maxDistance normalization. Build edge lookup Set for O(1) edge checking.
