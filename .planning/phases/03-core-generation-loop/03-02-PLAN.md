    ```typescript
    import type Anthropic from '@anthropic-ai/sdk';
    import { BUFFER_INTERVAL_MS } from '../types/settings';
    import type { SystemPromptBlock } from './prompt-builder';

    const WATCHDOG_TIMEOUT_MS = 30_000; // 30 seconds (GENP-11)

    export interface StreamResult {
      text: string;
      usage: { input_tokens: number; output_tokens: number };
      nodeContents: string[]; // Parsed from <node> delimiters
    }

    export interface StreamCallbacks {
      onFirstToken: () => void;              // Remove pulsing, set status to streaming
      onTextUpdate: (text: string) => void;  // Buffered text flush -- current node's visible content only
      onTimeout: () => void;                 // Watchdog fired (GENP-11)
      /**
       * Called when a </node> closing tag is detected mid-stream (D-03).
       * Provides the completed node's content (tags stripped) and its
       * zero-based index. The caller should use this to finalize the
       * current canvas node and pre-allocate the next one. After this
       * callback returns, subsequent onTextUpdate calls will contain
       * the NEW node's content (the caller should redirect updates to
       * the newly created canvas node).
       */
      onNodeBoundary: (completedNodeContent: string, nodeIndex: number) => void;
    }

    /**
     * Stream a Claude response with buffered updates and node boundary detection
     * (GENP-03, GENP-04, D-03).
     *
     * Key behavior for multi-node streaming (D-03):
     * - Accumulates raw text deltas from the API
     * - Tracks which <node> we're currently inside (nodeIndex)
     * - Strips <node>/<\/node> tags from the text flushed to onTextUpdate
     * - When </node> is detected mid-stream, calls onNodeBoundary with the
     *   completed node content and its index. This lets the caller finalize
     *   the current canvas node and create a new one.
     * - When the next <node> opens, resets the current-node buffer so
     *   onTextUpdate starts flushing content for the new node
     * - Flushes accumulated visible text to onTextUpdate at BUFFER_INTERVAL_MS
     * - Calls onFirstToken once when the first text delta arrives
     * - Watchdog fires onTimeout after 30s of no deltas (GENP-11)
     * - Respects AbortSignal for cancellation
     * - Returns accumulated text and usage after stream completes
     */
    export async function streamIntoNode(
      client: Anthropic,
      systemPrompt: SystemPromptBlock[],
      userMessage: string,
      signal: AbortSignal,
      callbacks: StreamCallbacks
    ): Promise<StreamResult> { ... }

    /**
     * Parse streamed response into individual node contents (D-02).
     * Splits on <node>...</node> delimiters. If no delimiters found,
     * treats entire text as a single node.
     */
    export function parseNodeContent(responseText: string): string[] { ... }
    ```

    Implementation details for `streamIntoNode`:
    - Call `client.messages.stream({ model: 'claude-opus-4-6', max_tokens: 4096, system: systemPrompt, messages: [{ role: 'user', content: userMessage }] }, { signal })`
    - Track state: `rawAccumulated` (full response), `currentNodeText` (visible text for current node, tags stripped), `nodeIndex` (0-based, which node we're streaming), `completedNodes: string[]` (content of finished nodes), `insideNode` (boolean), `lastFlush` timestamp, `firstTokenFired` boolean
    - In `.on('text', (delta) => { ... })`:
      1. Check `signal.aborted`
      2. Append delta to `rawAccumulated`
      3. Reset watchdog timer
      4. Call `onFirstToken` if first (once)
      5. **Tag-aware accumulation**: Process `rawAccumulated` for tag boundaries:
         - Scan the tail of `rawAccumulated` for `<node>` and `</node>` tags
         - When entering a `<node>`: set `insideNode = true`, reset `currentNodeText`
         - While inside a node: extract text between the opening `<node>` and current position (excluding any partial `</node>`), update `currentNodeText`
         - When a complete `</node>` is detected: call `callbacks.onNodeBoundary(currentNodeText.trim(), nodeIndex)`, push to `completedNodes`, increment `nodeIndex`, set `insideNode = false`, reset `currentNodeText`
      6. Flush `currentNodeText` to `onTextUpdate` if BUFFER_INTERVAL_MS elapsed since lastFlush
    - Handle partial tag detection: if the tail of `rawAccumulated` ends with a partial `<`, `</`, `</n`, `</no`, `</nod`, `</node`, or `<n`, `<no`, `<nod`, `<node` -- hold back from `currentNodeText` to avoid flashing tag fragments to the user. Only commit characters to `currentNodeText` once they are confirmed not part of a tag.
    - `await stream.finalMessage()` to get usage
    - Final flush: if any remaining `currentNodeText` not yet flushed, call `onTextUpdate`
    - Return `{ text: rawAccumulated, usage: { input_tokens, output_tokens }, nodeContents: parseNodeContent(rawAccumulated) }`

    Implementation details for `parseNodeContent`:
    - Regex: `/<node>([\s\S]*?)<\/node>/g`
    - Extract all matches, trim each
    - If no matches, return `[responseText.trim()]`
